-- 任务表
CREATE TABLE tasks (
    id INT PRIMARY KEY AUTO_INCREMENT,

    id INT PRIMARY KEY AUTO_INCREMENT,
    task_id INT NOT NULL,

### 6.2 异步执行实现

```python
import subprocess
import threading
import json
import time

class TestExecutor:
    def __init__(self, task_id, test_path, test_type, env_config):
        self.task_id = task_id
        self.test_path = test_path
        self.test_type = test_type
        self.env_config = env_config
        self.process = None
        self.log_lines = []

    def run(self):
        """在新线程中执行"""
        thread = threading.Thread(target=self._execute)
        thread.start()
        return thread

    def _execute(self):
        """执行测试"""
        cmd = self._build_command()

        # 设置环境变量
        env = os.environ.copy()
        env.update(self.env_config)

        # 启动进程
        self.process = subprocess.Popen(
            cmd,
            stdout=subprocess.PIPE,
            stderr=subprocess.STDOUT,
            env=env,
            shell=True
        )

        # 实时读取输出
        for line in iter(self.process.stdout.readline, ''):
            if line:
                line = line.decode('utf-8')
                self.log_lines.append(line)
                # 可通过WebSocket推送
                # ws.send(json.dumps({'type': 'log', 'data': line}))

        # 等待结束
        self.process.wait()

        # 生成报告
        self._generate_report()

        # 更新数据库
        self._update_execution()

    def _build_command(self):
        """根据测试类型构建命令"""
        if self.test_type == 'api':
            return f"pytest {self.test_path} -v --tb=short"
        elif self.test_type == 'ui':
            return f"pytest {self.test_path} -v --tb=short"
        elif self.test_type == 'performance':
            return f"locust -f {self.test_path} --headless -u 100 -r 10"
        else:
            return f"pytest {self.test_path} -v"
```

### 7.2 Jenkins API调用

```python
import requests
from requests.auth import HTTPBasicAuth

class JenkinsService:
    def __init__(self, url, username, token):
        self.url = url
        self.auth = HTTPBasicAuth(username, token)
        self.api = f"{url}/api/json"

    def get_jobs(self):
        """获取所有JOB"""
        response = requests.get(self.api, auth=self.auth)
        return response.json().get('jobs', [])

    def get_build_status(self, job_name, build_number):
        """获取构建状态"""
        url = f"{self.url}/job/{job_name}/{build_number}/api/json"
        response = requests.get(url, auth=self.auth)
        return response.json()

    def trigger_build(self, job_name, parameters=None):
        """触发构建"""
        url = f"{self.url}/job/{job_name}/build"
        if parameters:
            # 带参数构建
            url = f"{url}/WithParameters"
            response = requests.post(
                url,
                auth=self.auth,
                data=parameters
            )
        else:
            # 无参数构建
            response = requests.post(url, auth=self.auth)
        return response
```

### 8.2 禅道API调用

```python
class ZentaoService:
    def __init__(self, url, username, password):
        self.url = url
        self.username = username
        self.password = password
        self.session = None

    def login(self):
        """登录禅道"""
        response = requests.post(
            f"{self.url}/api.php/v1/session",
            json={'account': self.username, 'password': self.password}
        )
        if response.status_code == 200:
            self.session = response.json().get('session')
            return True
        return False

    def get_test_cases(self, product_id, module_id=None):
        """获取测试用例"""
        params = {'product': product_id}
        if module_id:
            params['module'] = module_id
        response = requests.get(
            f"{self.url}/api.php/v1/testcases",
            params=params,
            headers={'Z-Token': self.session}
        )
        return response.json()

    def create_bug(self, title, steps, severity, product_id):
        """创建Bug"""
        data = {
            'title': title,
            'steps': steps,
            'severity': severity,
            'product': product_id
        }
        response = requests.post(
            f"{self.url}/api.php/v1/bugs",
            json=data,
            headers={'Z-Token': self.session}
        )
        return response.json()
```
